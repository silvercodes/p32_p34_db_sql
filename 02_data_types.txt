=== Строковые типы ===
char(n)					fixed length, 		not Unicode, 	(n --> 1 - 8000)
varchar(n|MAX)			variable length, 	not Unicode 	(n --> 1 - 8000 | MAX)
nchar(n)				fixed length, 		Unicode, 		(n --> 1 - 8000)
nvarchar(n|MAX)			variable length, 	Unicode 		(n --> 1 - 8000 | MAX)

money					equals decimal(19, 4) 8 byte
